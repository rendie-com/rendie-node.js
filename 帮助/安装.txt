
npm config set registry https://registry.npmmirror.com          使用淘宝 (npmmirror) 镜像源
$env:PUPPETEER_DOWNLOAD_HOST="https://npmmirror.com/mirrors"    设置 Puppeteer 浏览器下载代理

npm init                                    生成 package.json（一路回车即可）

Remove-Item "$env:USERPROFILE\.cache\puppeteer" -Recurse -Force         # 删除 Puppeteer 的本地缓存
$env:PUPPETEER_SKIP_DOWNLOAD="true"                                     设置环境变量临时跳过下载
npx puppeteer browsers install chrome                                   # 运行 Puppeteer 自带的下载工具
node index.js

npm install dotenv


npm install puppeteer-extra puppeteer-extra-plugin-stealth             Puppeteer 反检测进阶指南
node --env-file=.env.local index.js                                    用这个可以加载.env.local运行
